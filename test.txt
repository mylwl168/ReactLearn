const [isModalOpen, setIsModalOpen] = useState(false);
const [modalData, setModalData] = useState({});



function handleCellDoubleClick(params) {
  const rowData = params.node.data;
  const url = `https://your-api-url/${rowData.id}`; // 根据单元格内容拼接API URL

  axios.get(url).then(response => {
    const data = response.data;
    setModalData(data); // 将返回的数据存储到状态中
    setIsModalOpen(true); // 打开弹出窗口
  });
}

<AgGridReact onCellDoubleClicked={handleCellDoubleClick} {...gridOptions} />



function Modal({ isOpen, data, onClose }) {
  if (!isOpen) {
    return null;
  }

  // 根据需要填充的数据渲染窗口内容
  return (
    <div className="modal">
      <div className="modal-content">
        <h2>{data.title}</h2>
        <p>{data.description}</p>
        {/* ... */}
      </div>
      <button onClick={onClose}>Close</button>
    </div>
  );
}
